CREATE TABLE `entries` (
	`playerOrTeamId` BIGINT(20) NOT NULL,
	`playerOrTeamName` VARCHAR(30) NULL DEFAULT NULL,
	`division` VARCHAR(20) NULL DEFAULT NULL,
	`leaguePoints` INT(11) NULL DEFAULT NULL,
	`wins` INT(11) NULL DEFAULT NULL,
	`losses` INT(11) NULL DEFAULT NULL,
	`playstyle` VARCHAR(30) NULL DEFAULT 'NONE',
	`isHotStreak` TINYINT(1) NULL DEFAULT NULL,
	`isVeteran` TINYINT(1) NULL DEFAULT NULL,
	`isFreshBlood` TINYINT(1) NULL DEFAULT NULL,
	`isInactive` TINYINT(1) NULL DEFAULT NULL,
	PRIMARY KEY (`playerOrTeamId`)
)

CREATE TABLE `league` (
	`id` BIGINT(20) NOT NULL,
	`name` VARCHAR(30) NULL DEFAULT NULL,
	`tier` VARCHAR(30) NULL DEFAULT NULL,
	`queue` VARCHAR(30) NULL DEFAULT NULL,
	`entries` BIGINT(20) NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)

CREATE TABLE `summoner` (
	`id` BIGINT(20) NOT NULL,
	`revisionDate` BIGINT(20) NULL DEFAULT NULL,
	`summonerLevel` INT(11) NULL DEFAULT NULL,
	`profileIconId` INT(11) NULL DEFAULT NULL,
	`name` VARCHAR(40) NULL DEFAULT NULL,
	PRIMARY KEY (`id`)
)
